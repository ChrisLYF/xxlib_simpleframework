	{
		std::cout << "err code = " << rtv << ", err msg = " << lua_tostring(scene.L, -1) << std::endl;

	}

	{
		std::cout << "err code = " << rtv << ", err msg = " << scene.err.C_str() << std::endl;

	}
	return 0;
